JavaScript: Classes

Important: Access to instance members of a class always with this. 


              class Person {
                constructor(name) {
                    this._name = name;
                }
                
                get name() {
                    return this._name;
                }
                
                set name(newName) {
                    this._name = newName;
                }
                
                toString() { return "Person: " + this.name + ""; }
            }
            
            const p = new Person("Susi");
            p.toString() // Person: Susi
          

              p.name // Susi (getter)
              p.name = "Lisa"; // setter

              console.log(p.toString()) // Person: Lisa;
            

              p._name // Lisa (direct field access, no visibility restrictions!)  üò±
              p._name = "Moni"; // (direct field access, no visibility restrictions!) üò±
              console.log(p.toString()) // Person: Moni;

JavaScript: Classes #2


class Musician extends Person {
  constructor(name, instrument) {
    // call to 'super' always must be first expression!
    super(name);
    this._instrument = instrument;
  }

  toString() { return super.toString() + " plays " + this._instrument}
}            

const lemmy = new Musician("Lemmy", "Bass");
lemmy.toString(); // Lemmy plays Bass
lemmy.name // // Lemmy

Modules

Named exports

Beside the (one) default export, a module can contain one or more named exports 


                // src/domain/Person.js
                export default class Person { ...}
                export function createPerson() { ... }
                export function removePerson() { ... }
              

                  // src/domain/Person.js
                  export default class Person { ...}
                  function createPerson() { ... }
                  function removePerson() { ... }

                  export { createPerson, removePerson }

Modules

Names imports


                    // src/App.js
                    import { createPerson } from "./Person.js"
              

                // Import default and named imports together
                import Person, { createPerson } from "./Person.js"
              

                  // Import everything
                  import { * as P } from "./Person.js"
  
                  P.createPerson();
                

                    // Rename in import (for example to avoid naming clashes)
                    import { createPerson as create } from "./Person.js"
    
                    create();
